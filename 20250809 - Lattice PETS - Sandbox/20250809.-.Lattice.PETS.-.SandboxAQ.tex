\section{The Trouble}
\label{sec:orgb5c062b}
\begin{frame}[label={sec:org68aef1e}]{Trapdoor Friendly}
\begin{alertblock}{The Problem}
The server has to output \(c_{x} \cdot \alert{k} + \alert{e}'\) for some \(c_x \overset{?}{=}  H(x) + a \cdot \alert{r} + \alert{e}\). This is not safe, the client can learn all the server's secrets with one maliciously crafted \(c_{x}\).
\end{alertblock}
\begin{itemize}
\item The client proves in zero-knowledge that \(c_{x}\) is well-formed: \(c_{x} \coloneqq H(x) + a \cdot \alert{r} + \alert{e}\)
\item This is expensive in terms of bandwidth and computation cost
\begin{itemize}
\item \cite{PKC:ADDS21}: \(\approx\) 128GB per evaluation using \cite{C:YAZXYW19}
\item \cite{AC:AlbGur24}: \(\approx\) 63kB per evaluation using \cite{C:BeuSei23}
\end{itemize}
\end{itemize}
\end{frame}
\begin{frame}[label={sec:org7e14751}]{Noise Leakage}
\begin{alertblock}{The Problem}
The client learns \(\alert{e}\cdot\alert{k} + \alert{e}'  - \alert{e}'' \cdot \alert{r}\)
where it chooses \(\alert{e}\) and \(\alert{r}\). 
\end{alertblock}
\begin{itemize}
\item Drown everything with \(\alert{e'}\): \(\Vert\alert{e}'\Vert \gg  \Vert\alert{e}\cdot\alert{k}  - \alert{e}'' \cdot \alert{r}\Vert\)
\item Since \(q > \Vert\alert{e}'\Vert\) we have that \(q/\Vert \alert{e} \Vert\) -- the ``signal to noise ratio'' -- is big.
\item A big signal to noise ratio makes solving LWE easier.
\item This requires us to use larger secret dimensions \(n\) to compensate.
\item Bandwidth cost is essentially \(n \log q\).
\end{itemize}
\end{frame}
\begin{frame}[label={sec:orgf9aa4eb}]{Noise Growth}
\begin{columns}
\begin{column}{0.65\columnwidth}
\begin{alertblock}{The Problem}
Have:
\[
H(x)\cdot\alert{k} + \alert{e}\cdot\alert{k} + \alert{e'}  - \alert{e''} \cdot \alert{r}
\approx H(x)\cdot\alert{k}.
\]

\begin{frame}[label={sec:orgbe80f04}]{An Alternative from FHE}
We do have efficient FHE: ciphertexts are typically \alert{smaller} than messages above.
\begin{itemize}
\item \textbf{Simple idea:}
\begin{enumerate}
\item Client FHE encrypts \(x\) as \([x]\)
\item Server homomorphically computes PRF using plaintext \(k\) and \([x]\) to obtain \([F_{k}(x)]\)
\item Client FHE decrypts \(F_{k}(x)\)
\end{enumerate}
\item \textbf{Problem:} PRFs need deep circuits, deep circuits are expensive
\item \textbf{Proposal:} Use Dark Matter (weak-)PRF candidate \cite{TCC:BIPSW18} \(\sum \left(\mat{A}\cdot \vec{x} \bmod 2\right) \bmod 3\) where \(\mat{A}\) is the secret key
\end{itemize}
